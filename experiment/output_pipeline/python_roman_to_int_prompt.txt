def roman_to_int(s):
    roman_map = {'I': 1, 'V': 5, 'X': 10, 'L': 50, 'C': 100, 'D': 500, 'M': 1000}
    if not s:
        return 0
    total = 0
    n = len(s)
    for i in range(n):
        if s[i] not in roman_map:
            return 0
        val = roman_map[s[i]]
        if i + 1 < n and roman_map[s[i + 1]] > val:
            total -= val
        else:
            total += val
    return total
