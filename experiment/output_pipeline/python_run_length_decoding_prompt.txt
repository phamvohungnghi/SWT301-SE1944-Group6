def run_length_decoding(s):
    if not s:
        return ""
    decoded = []
    i = 0
    while i < len(s):
        if not s[i].isdigit():
            return ""
        num_str = ""
        while i < len(s) and s[i].isdigit():
            num_str += s[i]
            i += 1
        if i >= len(s):
            return ""
        char = s[i]
        decoded.append(char * int(num_str))
        i += 1
    return "".join(decoded)
