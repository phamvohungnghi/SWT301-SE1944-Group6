def parse_query_string(qs):
    if not qs:
        return {}
    if qs.startswith("?"):
        qs = qs[1:]
    result = {}
    pairs = qs.split("&")
    for pair in pairs:
        if not pair:
            continue
        if "=" in pair:
            parts = pair.split("=", 1)
            key, val = parts[0], parts[1]
        else:
            key, val = pair, ""
        if key in result:
            if isinstance(result[key], list):
                result[key].append(val)
            else:
                result[key] = [result[key], val]
        else:
            result[key] = val
    return result
