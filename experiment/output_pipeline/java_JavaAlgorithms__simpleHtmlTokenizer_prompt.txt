        while (i < n) {

                    j++;

                if (j < n) {

                    i = j + 1;
                } else {
                    tokens.add(new String[]{"TEXT", html.substring(i)});
                    break;
                }
            } else {
                int j = i;
                while (j < n && html.charAt(j) != '<') {
                    j++;
                }
                tokens.add(new String[]{"TEXT", html.substring(i, j)});
                i = j;
            }
        }
        return tokens;
    }
