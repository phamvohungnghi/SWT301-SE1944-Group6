def simple_html_tokenizer(html):
    if html is None:
        return []
    tokens = []
    i = 0
    n = len(html)
    while i < n:
        if html[i] == '<':
            j = i + 1
            while j < n and html[j] != '>':
                j += 1
            if j < n:
                tokens.append(("TAG", html[i:j+1]))
                i = j + 1
            else:
                tokens.append(("TEXT", html[i:]))
                break
        else:
            j = i
            while j < n and html[j] != '<':
                j += 1
            tokens.append(("TEXT", html[i:j]))
            i = j
    return tokens
