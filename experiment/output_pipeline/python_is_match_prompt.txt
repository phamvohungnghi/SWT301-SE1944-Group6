def is_match(s, p):
    if p is None or s is None:
        return False
    if not p:
        return not s
    first_match = bool(s) and p[0] in (s[0], '.')
    if len(p) >= 2 and p[1] == '*':
        return (is_match(s, p[2:]) or
                (first_match and is_match(s[1:], p)))
    else:
        return first_match and is_match(s[1:], p[1:])
