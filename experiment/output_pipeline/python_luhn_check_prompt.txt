def luhn_check(card_number):
    if not card_number or not card_number.isdigit():
        return False
    total = 0
    reverse_digits = card_number[::-1]
    for idx, digit in enumerate(reverse_digits):
        val = int(digit)
        if idx % 2 == 1:
            val *= 2
            if val > 9:
                val -= 9
        total += val
    return total % 10 == 0
